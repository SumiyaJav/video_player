### Installation
git clone https://github.com/SumiyaJav/video_player.git
cd video_player
npm install
npm start


src/
├── pages/
│   ├── Home.js          # Home grid of videos
│   ├── Create.js        # Video creation form
│   └── VideoPlayer.js   # Watch and comment on videos
├── App.js
├── index.js
├── assets/              # Backgrounds and images
├── tailwind.config.js
├── postcss.config.js
